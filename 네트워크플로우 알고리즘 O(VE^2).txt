#include <iostream>
#include <vector>
#include <queue>
#include <algorithm>

#define MAX 100
#define INF 1000000000

using namespace std;

// 에드몬드 카프알고리즘
// 네트워크 플로우 알고리즘
// 기본적으로 BFS 사용해서 더 사용가능한 부분까지 찾아서 사용

int n = 6, result;
int c[MAX][MAX], f[MAX][MAX], d[MAX];
vector<int> a[MAX];

void maxFlow(int start, int end) {
	while (1)
	{
		fill(d, d + MAX, -1);
		queue<int> q;
		q.push(start);
		while (!q.empty()) {
			int x = q.front();
			q.pop();
			for (int i = 0; i < a[x].size(); i++)
			{
				int y = a[x][i];
				//방문하지 않은 노드 중에 용량이 남은 경우
				if (c[x][y] - f[x][y] > 0 && d[y] == -1) {
					q.push(y);
					d[y] = x;
					if (y == end) break;
				}

			}
		}
		//모든 경로를 찾은 뒤에 탈출
		if (d[end] == -1) break;
		int flow = INF;
		// 거꾸로 최소 유량 탐색
		for (int i = end; i != start ; i = d[i])
		{
			flow = min(flow, c[d[i]][i] - f[d[i]][i]);
		}
		// 최소 유량만큼 추가
		for (int i = end; i != start; i = d[i])
		{
			f[d[i]][i] += flow;
			f[i][d[i]] -= flow;
		}
		result += flow;
	}
}

	a[2].push_back(1);

	c[2][5] = 5;

	a[2].push_back(6);

	c[2][6] = 9;

	a[3].push_back(6);

}
