#include <stdio.h>
#include <math.h>

//제곱근 이용한 방식 O(N)/2
bool isPrimeNumber(int x) 
{
	int end = (int)sqrt(x);
	for (int i = 2; i <= end; i++)
	{
		if (x % i == 0) return false;
	}
	return true;
}
