#include <iostream>
#include <stack>
#include <vector>

using namespace std;

int number = 7;
int c[7]; //checked
vector<int> a[8];

void dfs(int x) 
{
	if (c[x]) return;	// 방문 했으면 리턴
	c[x] = true;		// 방문 true 찍고
	cout << x << ' ';   // x 출력
	for (int i = 0; i < a[x].size(); i++)	//정점의 크기만큼
	{
		int y = a[x][i];
		dfs(y); //재귀
	}
}


int main(void) 
{
	a[1].push_back(2);
	a[2].push_back(1);

	a[3].push_back(7);
	a[7].push_back(3);

	dfs(1);

	return 0;
}
